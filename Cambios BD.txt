Cambios para modulo tiempoXtiempo
CREATE TABLE timebytime (
    id INT PRIMARY KEY AUTO_INCREMENT,
    usuario_id INT,
    folio varchar(10),
    archivo LONGBLOB,
    estatus ENUM('pendiente', 'entregado') NOT NULL DEFAULT 'pendiente';
    fechaR DATE DEFAULT (CURRENT_DATE),
    FOREIGN KEY (usuario_id) REFERENCES usuario(usuario_id) ON DELETE CASCADE
);

CREATE TABLE timebytimefaltas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    timebytime_id INT,
    fechaF DATE,
    horasF INT,
    FOREIGN KEY (timebytime_id) REFERENCES TimeByTime(id) ON DELETE CASCADE
);

CREATE TABLE timebytimepagos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    timebytime_id INT,
    fechaP DATE,
    horaP INT,
    estatusP TINYINT NOT NULL DEFAULT 1,
    FOREIGN KEY (timebytime_id) REFERENCES TimeByTime(id) ON DELETE CASCADE
);

Modificacion Para El Modulo De Comiciones

CREATE TABLE Transporte (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    marca VARCHAR(50),
    modelo VARCHAR(50),
    color VARCHAR(50),
    Placas VARCHAR(15)
);

CREATE TABLE Comiciones (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    Fecha_de_Elaboracion DATE,
    Usuario_Id INT,
    Lugar VARCHAR(255),
    Asunto VARCHAR(100),
    Viaticos ENUM('Si', 'No'),
    Especificacion_Viaticos VARCHAR(100),
    Observaciones VARCHAR(100),
    Fecha_De_Salida DATETIME,
    Fecha_De_Regreso DATETIME,
    Transporte_propio ENUM('Si', 'No'),
    marca VARCHAR(50),
    modelo VARCHAR(50),
    color VARCHAR(50),
    Placas VARCHAR(15),
    Transporte INT,
    kilometraje INT NULL,
    Status VARCHAR(50)
);
